tree(tree(A,Prs))       --> number(X), "x", number(Y), ": ", sequence(number," ",Prs), { #A #= #X * #Y }.
input(Trees)            :- phrase_from_file((...,sequence(tree,"\n",Trees)),"i/12.in").

solve(P1,":D")          :- input(Trees), tfilter(solvable,Trees,STrees), length(STrees,P1).
